first_rashudin_invasion = {
	name = CB_NAME_RASHUDIN_INVASION

	sprite = 8
	truce_days = 0
	hostile_against_others = yes
	allow_whitepeace = no
	is_permanent = no
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	allowed_to_target_tributaries = yes
	
	sort_priority = 100000

	can_use = {
		ROOT = {
			has_landed_title = e_arabia
		}
	}
	
	is_valid = {
		always = yes
	}

	on_success = {
		FROM = {
			show_scope_change = no
			prestige = -200
		}
	}
	
	on_success_title = {
		custom_tooltip = {
			text = other_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					gain_settlements_under_title = {
						title = PREV
						enemy = FROM
						type = invasion
					}
					usurp_title = PREV
					decadence = -10
				}
				while = {
					count = 5
					ROOT = {
						random_realm_province = {
							limit = {
								de_jure_liege_or_above = PREVPREV
								OR = {
									NOT = { culture = ROOT }
									NOT = { religion = ROOT }
								}
							}
							culture = ROOT
							religion = ROOT
						}
					}
				}
			}
			if = {
				limit = { FROM = { ai = no } }
				FROM = { set_defacto_liege = ROOT }
			}
			ROOT = { character_event = { id = HFP.41090 days = 1 } }
		}
	}
	
	on_fail = {
		FROM = {
			show_scope_change = no
			prestige = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		ROOT = {
			show_scope_change = no
			prestige = -200
		}
	}
	
	on_invalidation = {
		custom_tooltip = {
			text = other_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					gain_settlements_under_title = {
						title = PREV
						enemy = FROM
						type = invasion
					}
					usurp_title = PREV
					decadence = -10
				}
				while = {
					count = 20
					ROOT = {
						random_realm_province = {
							limit = {
								de_jure_liege_or_above = PREVPREV
								OR = {
									NOT = { culture = ROOT }
									NOT = { religion = ROOT }
								}
							}
							culture = ROOT
							religion = ROOT
						}
					}
				}
			}
			if = {
				limit = { FROM = { ai = no } }
				FROM = { set_defacto_liege = ROOT }
			}
			
			ROOT = { character_event = { id = ROMA.999 days = 1 } }
			ROOT = { character_event = { id = HFP.41090 days = 1 } }
		}
	}

	on_reverse_demand = {
		FROM = {
			c_dhu_zabi = {
				grant_title = PREV
			}
			c_hajar = {
				grant_title = PREV
			}
			c_muscat = {
				grant_title = PREV
			}
			c_duqm = {
				grant_title = PREV
			}
		}
		d_arabia_petrae = {
			holder_scope = {
				c_hijaz = {
					grant_title = PREV
				}
			}
		}
		custom_tooltip = {
			text = byz_rashudin_ortho
			hidden_tooltip = {
				create_character = {
					dynasty=random
					age = 33
					attributes = {
						martial=9
						diplomacy=8
						intrigue=5
						stewardship=7
					}
					health = 6
					fertility = 0.7
					religion="orthodox"
					culture="bedouin_arabic"
					random_traits = yes
				}
				new_character = {
					gain_settlements_under_title = {
						title = k_arabia
						enemy = ROOT
						type = invasion
					}
					usurp_title = k_arabia
					wealth = 500
					prestige = 500
					piety = 500
					save_event_target_as = new_ruler
					while = {
						count = 2
						spawn_unit = {
							province = 840 # Fes
							attrition = 0
							troops = {
								light_infantry = { 2000 2000 }
								heavy_infantry = { 500 500 }
								archers = { 1000 1000 }
							}
						}
					}
				}
				while = {
					count = 5
					k_arabia = {
						holder_scope = {
							random_realm_province = {
								limit = {
									de_jure_liege_or_above = PREVPREV
									OR = {
										NOT = { culture = bedouin_arabic }
										NOT = { religion = orthodox }
									}
								}
								culture = bedouin_arabic
								religion = orthodox
							}
						}
					}
				}
			}
			k_arabia = {
				holder_scope = {
					d_sanaa = {
						grant_title = PREV
					}
					c_bayda = {
						grant_title = PREV
					}
					c_aden = {
						grant_title = PREV
					}
				}
			}
			e_byzantium = {
				holder_scope = {
					make_tributary = { who = event_target:new_ruler tributary_type = permanent }
				}
			}
			if = {
				limit = { FROM = { ai = no } }
				FROM = { set_defacto_liege = ROOT }
			}
			ROOT = { character_event = { id = HFP.41090 days = 1 } }
			hidden_effect = { any_player = { narrative_event = { id = RASHD.200 days = 1 } } }
		}
		while = {
			count = 20
			FROM = {
				random_realm_province = {
					limit = {
						de_jure_liege_or_above = PREVPREV
					}
					religion = orthodox
				}
			}
		}
		e_arabia = {
			holder_scope = {
				any_demesne_title = {
					limit = {
						title = e_arabia
					}
					unsafe_destroy_landed_title = THIS
				}
				any_demesne_title = {
					limit = {
						title = d_sunni
					}
					unsafe_destroy_landed_title = THIS
				}
			}
		}
