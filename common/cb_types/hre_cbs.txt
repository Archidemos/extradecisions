hre_imperial_trib_cb = {
	name = CB_NAME_IMPERIAL_TRIB
	war_name = WAR_NAME_IMPERIAL_TRIB
	sprite = 37
	truce_days = 7300
	is_permanent = yes
	hostile_against_others = yes
	can_ask_to_join_war = yes
	display_on_map = no
	allowed_to_target_tributaries = yes
	
	defender_unoccupied_warscore = yes
	
	sort_priority = 820
	
	can_use_gui = {
		check_if_crusader_trigger = yes
		ROOT = { has_landed_title = e_hre }
	}
	
	can_use = {
		ROOT = {
			has_game_rule = {
				name = shatter_hre_rule
				value = on
			}

			has_landed_title = e_hre
			independent = yes
			has_horde_culture = no
			is_nomadic = no
			mercenary = no
		}

			any_realm_province = {
				location = {
					region = world_europe_west_germania
				}
			}
			#NOR = { 
			#	tier = KING
			#	tier = EMPEROR
			#}
			NOT = { tier = EMPEROR }
			NOT = { is_offmap_governor = yes }
			rebel = no
		}
	}

	is_valid = {
		NOT = { ROOT = { rightful_religious_head_scope = { character = FROM } } }
	}

	on_add = {
		# Chance to surrender early
		ROOT = { save_event_target_as = tribute_war_suzerain }
		FROM = {
			show_scope_change = no
			save_event_target_as = tribute_war_target
			set_character_flag = surrender_tributary_client_cb
			character_event = { id = JD.2500 days = 5 tooltip = surrender_tributary_tt }
		}
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }
			attacker = {  
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes 
			}
		}
	}

	on_success = {
		FROM = {
			show_scope_change = no
			save_event_target_as = target_loser
			prestige = -400
			custom_tooltip = {
				text = "REMOVE_PREVIOUS_SUZERAIN"
				hidden_tooltip = {
					any_suzerain = {
						event_target:target_loser = {
							remove_tributary = PREV
						}
					}
				}
			}
			opinion = {
				modifier = opinion_forced_tributary 
				who = ROOT
				years = 15
			}
		}

		ROOT = {
			show_scope_change = no
			prestige = 200
			if = {
				limit = { NOT = { has_character_modifier = recently_released_vassal } }
				add_character_modifier = {
					modifier = recently_released_vassal
					years = 5
					inherit = yes
				}
			}
			make_tributary = { who = FROM tributary_type = hre_imperial }
		}
		
		if  = {
			limit = {
				ROOT = { uses_decadence = yes }
			}
			ROOT = { decadence = -10 }
		}
		if  = {
			limit = {
				ROOT = { is_nomadic = yes }
			}
			ROOT = {
				clan_title = {
					ROOT = {
						any_vassal = {
							limit = { is_nomadic = yes }
							clan_title = {
								clan_opinion = {
									who = PREVPREVPREV
									modifier = won_tributary_war
									years = 5
								}
							}
						}
					}
				}
			}
		}
		
		if = {
			limit = {
				ROOT = {
					is_tribal = yes
				}
			}
			ROOT = {
				any_vassal = {
					limit = { is_tribal = yes }
					opinion = {
						who = PREV
						modifier = won_tributary_war
						years = 5
					}
				}
			}
		}
	}

	on_fail = {
		ROOT = {
			show_scope_change = no
			prestige = -1000
		}
		FROM = {
			show_scope_change = no
			prestige = 1000
		}
	}

	on_reverse_demand = {
		ROOT = {
			show_scope_change = no
			prestige = -2000
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}

		FROM = {
			show_scope_change = no
			prestige = 2000
		}
	}
	
	on_attacker_leader_death = {
		end_war = invalid
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = {
		factor = 1
		
		modifier = {
			factor = 20  # High prio CB
		}
		modifier = { #Focus on Coronation first.
			factor = 0.1
			has_dlc = "Holy Fury"
			ROOT = {
				is_feudal = yes
				OR = {
					religion = catholic
					religion = fraticelli
				}
				NOR = {
					trait = crowned_by_priest
					trait = crowned_by_bishop
					trait = crowned_by_pope
					trait = crowned_by_myself
				}
			}
		}
	}
}
