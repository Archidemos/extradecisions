	custom_tooltip = {
		text = tt_old_german_dejure
		
		e_hre = {
			any_direct_de_jure_vassal_title = {
				de_jure_liege = e_germany
			}
		}
		
		## Bohemia
		if = {
			limit = {
				is_title_active = e_wendish_empire
			}
			k_bohemia = { de_jure_liege = e_wendish_empire }
		}
		else_if = {
			limit = {
				is_title_active = e_lechczechrus
			}
			k_bohemia = { de_jure_liege = e_lechczechrus }
		}
		else = {
			k_bohemia = { de_jure_liege = e_germany }
		}
		
		## Saxony
		activate_title = { #deactivate the "grand duchy" titles and use ottonian kingdoms instead
			title = k_saxon
			status = no
		}
		
		activate_title = {
			title = k_saxony
			status = yes
		}
		
		k_saxony = { de_jure_liege = e_germany }
		
		d_saxony = { de_jure_liege = k_saxony }
