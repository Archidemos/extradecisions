character_event = { #Pope event to trigger the first Player Event
	id = AZT.96
	hide_window = yes
	is_triggered_only = yes
	
	only_playable = yes
	religion = catholic
	min_age = 16
	
	trigger = {
		OR = {
			AND = {
				has_game_rule = {
					name = aztec_invasion
					value = random
				}
				total_years_played = 50
			}
			AND = {
				has_game_rule = {
					name = aztec_invasion
					value = delayed_random
				}
				year >= 1000
				total_years_played = 50
			}
			AND = {
				has_game_rule = {
					name = aztec_invasion
					value = test
				}
				total_years_played <= 1
			}
		}
		AND = {
			has_game_rule = {
				name = aztec_invasion
				value = test
			}
			year = 769
		}
		NOT = { has_global_flag = aztec_part_1_triggered }
		NOT = { has_global_flag = aztec_arrival }
		NOR = {
			has_game_rule = {
				name = aztec_invasion
				value = off
			}
			has_global_flag = demon_child_to_aztecs
		}
		controls_religion = yes # Let it happen for the Pope. He's always around.
	}
	
	immediate = {
		any_independent_ruler = {
			limit = {
				OR = {
					culture = saxon
					culture = english
					culture = scottish
					culture = cumbric
					culture = anglonorse
				}
			}
			save_event_target_as = random_britanian_spotted_aztec
		}
		any_player = { narrative_event = { id = AZT.97 } }
		ROOT = { character_event = { id = AZT.98 days = 30 } }
		set_global_flag = aztec_part_1_triggered
	}
}

				set_character_flag = is_aztec_invader

					set_title_landless = { title = THIS status = yes }

				}

					province = 1740 # Land's End
					troops = {
						galleys = { 1000 1000 }
					}
				}
			}
		}
		32 = {
			kingdom = {
				e_mexikha = {
					holder_scope = {
						unsafe_war = {
							casus_belli = tribal_invasion

							thirdparty_title = PREVPREV
						}
					}
				}
			}
		}
		e_mexikha = { holder_scope = { culture = nahuatl } }
	}

		name = "EVTOPTA60500"
		trigger = {
			OR = {
				religion_group = christian

			}
