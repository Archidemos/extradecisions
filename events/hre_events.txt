character_event = {
	id = HREF.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		year >= 935
		has_game_rule = {
			name = shatter_hre_rule
			value = on
		}
		ROOT = { has_landed_title = e_hre }
	}
	
	immediate = {
		current_heir = { character_event = { id = HREF.2 days = 1 } }
	}
}


character_event = {
    id = HREF.2

	title = "Imperial Succession"
	desc = "Administrating this massive empire is no easy task and so we have no choice but to give our inherited subjects their semi-independence as a Imperial Realm."

	picture = "GFX_evt_hre_eagle"
	border = GFX_event_normal_frame_diplomacy
	
	is_triggered_only = yes

	option = {
		custom_tooltip = { text = "It shall be done" }
		ai_chance = { factor = 100 }

				AND = {

				}
			}

		}
		hidden_tooltip = {
			set_government = hre_realm_government
		}
	}
}

character_event = { #France converts appanages to right title
    id = HREF.201
	hide_window = yes
	is_triggered_only = yes
	
    trigger = {
		year >= 935
		has_game_rule = {
			name = shatter_hre_rule
			value = on
		}
		is_tributary = { type = hre_imperial }
		NOT = {
			has_opinion_modifier = {
				modifier = hre_loyalty
				who = ROOT
			}
		}
    }

    immediate = {
		e_hre = {
			holder_scope = {
				save_event_target_as = hre_friend
				#add_friend = ROOT
				opinion = {
					name = hre_loyalty
					who = ROOT
					years = 100
				}
			}
		}
		opinion = {
			name = hre_loyalty
			who = event_target:hre_friend
			years = 100
		}
	}
}
