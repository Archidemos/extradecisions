namespace = BEX

character_event = {
	id = BEX.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		in_command = yes
		is_alive = yes
	}
	
	immediate = {
		change_variable = {
			which = experienced_commander
			value = 1
		}

		if = {
			limit = { check_variable = { which = experienced_commander value == 5 } }
			add_trait = battle_expierenced_5
		}
		else_if = {
			limit = { check_variable = { which = experienced_commander value == 15 } }
			add_trait = battle_expierenced_10
			remove_trait = battle_expierenced_5
		}
		else_if = {
			limit = { check_variable = { which = experienced_commander value == 25 } }
			add_trait = battle_expierenced_15
			remove_trait = battle_expierenced_10
		}
		else_if = {
			limit = { check_variable = { which = experienced_commander value == 35 } }
			add_trait = battle_expierenced_20
			remove_trait = battle_expierenced_15
		}
		else_if = {
			limit = { check_variable = { which = experienced_commander value == 45 } }
			add_trait = battle_expierenced_25
			remove_trait = battle_expierenced_20
		}
	}
}
