namespace = SELJEX

character_event = {
	id = SELJEX.1

	hide_window = yes
    is_triggered_only = yes

	trigger = {
		log = "SELJEX.1 Triggered"
		FROM = {
			ai = yes
			NOT = { any_war = { count = 5 } }
			has_landed_title = e_seljuk_turks
			has_character_flag = seljuk_himself
			NOT = { completely_controls_region = world_central_persia }
			NOT = { has_global_flag = world_central_persia_conquered_sel }
		}
	}
	
	immediate = {
		FROM = { character_event = { id = SELJEX.2 days = 2 } }
		FROM = { character_event = { id = SELJEX.2 days = 2 } }
		log = "SELJEX.1 Fired"
	}
}

character_event = { #Timur being guided to conquer Persia
	id = SELJEX.2
	title = "Seljuk Expansion"
	desc = "Onwards to our next target in Persia"

    is_triggered_only = yes
	
	picture = GFX_evt_mongols
	border = GFX_event_normal_frame_war
	
	option = {
		name = "Onwards!"
		log = "SELJEX.2 Fired"
		change_infamy = {
			value = -30
			localisation = "Threat Reduced with the Decision"
		}
		random_province = {
			log = "SELJEX.2 Option Works"
            limit = {
                has_owner = yes
                region = world_central_persia
                owner = {
                    NOR = {
                        character = ROOT
                        is_liege_or_above = ROOT
                    }
                    top_liege = {
                        reverse_has_cb = ROOT
                    }
                }
				any_neighbor_province = {
					owner = {
						OR = {
							character = ROOT
							is_liege_or_above = ROOT
						}
					}
				}
            }
			top_liege = {
				ROOT = {
					unsafe_war = {
						casus_belli = seljuk_special_cb
						target = PREV
						thirdparty_title = PREVPREV
						tier = KING
					}
				}
			}
		}
	}
}

				create_character = {

					trait="ambitious"
					trait="diligent"
					trait="cavalry_leader"
					trait="brilliant_strategist"
					trait="great_conquerer"
				}
				new_character = {
					set_character_flag = rum_himself
					set_character_flag = is_special_character
					gain_settlements_under_title = {
						title = k_nikaea
						enemy = ROOT
						type = invasion
					}
					gain_settlements_under_title = {
						title = k_anatolia
						enemy = ROOT
						type = invasion
					}
					gain_settlements_under_title = {
						title = k_trebizond
						enemy = ROOT
						type = invasion
					}
					create_bloodline = {
						type = seljuk
					}
					k_rum = {
						set_title_landless = { title = THIS status = yes }
						grant_title = PREV
						add_law = succ_turkish_succession
					}
					set_government_type = muslim_government
					wealth = 5000
					prestige = 2500
					piety = 1500
					add_claim = d_thrace
					add_claim = c_byzantion
					set_character_flag = rum_invader
					while = {
						count = 3
						spawn_unit = {
							province = 759 # Fes
							#attrition = 1.0
							troops = {
								light_infantry = { 2000 2000 }
								heavy_infantry = { 500 500 }
								archers = { 1500 1500 }
								light_cavalry = { 500 500 }
								camel_cavalry = { 500 500 }
							}
						}
					}
					character_event = { id = RUME.101 } # Spawn Temujins generals
					character_event = { id = RUME.102 } # Spawn courtiers
				}
			}
		}
		any_player = { narrative_event = { id = SELJEX.12 days = 2 } }
	}

	option = {
		name = "We keep these lands under us"
		ai_chance = { factor = 0 }
		set_character_flag = spawned_rum
	}
}

	portrait = e_seljuk_turks

	border = GFX_event_narrative_frame_war
	
	trigger = {
		multiplayer = no
		ai = no

	}

	immediate = { e_seljuk_turks = { holder_scope = { save_event_target_as = play_as_seljuk } } }

		set_global_flag = sejluk_playable
	}
	
	option = {
		name = "Let's not get distracted!"
		set_global_flag = sejluk_playable
	}
