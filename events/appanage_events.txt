letter_event = {
	id = APP.2
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		any_playable_ruler = {
			limit = { is_tributary = { type = appanage } }
			letter_event = { id = APP.3 days = 1 }
		}
	}
}

letter_event = {
	id = APP.3
	desc = "My Appanage, as your previous Roi has passed away. I demand your oath of fealy once more as your new Roi of France and i therefore expect you in the capital at once."
	is_triggered_only = yes
	
	trigger = { is_tributary = { type = appanage } }
	
	immediate = {
		k_france = { holder_scope = { save_event_target_as = roi_france } }
	}

	option = {
		name = "We shall remain an Appanage"
		ai_chance = { factor = 100 }
		custom_tooltip = { text = "You will remain as an Appanage tributary." }
	}
	
	option = {
		name = "We need to swear fealty to the Roi"
		ai_chance = { factor = 0 }
		set_defacto_liege = event_target:roi_france
	}
	
	option = {
		name = "We will declare independence"
		ai_chance = { factor = 0 }
		suzerain = { remove_tributary = ROOT }
		k_france = { holder_scope = { letter_event = { id = APP.4 days = 5 } } }
	}
}
