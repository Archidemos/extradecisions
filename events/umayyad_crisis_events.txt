character_event = {
	id = UMM.10
    hide_window = yes
    is_triggered_only = yes
	
    trigger = {
		start_date <= 936.8.7
		is_save_game = no
		NOT = { has_global_flag = umayyad_player_event_fired }
    }
	
	immediate = {
		set_global_flag = umayyad_player_event_fired
		any_player = { narrative_event = { id = UMM.6 days = 5 } }
		k_andalusia = {
			save_event_target_as = character_umayyad_crisis
		}
	}
}
