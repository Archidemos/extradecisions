namespace = AGE

narrative_event = {
	id = AGE.1
	title = "The Dark Ages"
	desc = "As Christianity is on the rise while the Roman Empire is on the decline. Many wars happend after, from migrations to kingdoms that fell. As kingdoms fall, so do cities and civilisations and their ancient cities and foritifications as well as important history and knowledge of the old world. While these times are dark and grim, it could also be seen as a clense towards a new age and a new world. Perhaps a new chance for society and civilisation for the coming ages."
	major = yes
	#hide_from = yes
    is_triggered_only = yes
	only_independent = yes
	
	picture = GFX_evt_burning_castle_hf
	border = "GFX_event_narrative_frame_war"
	sound = ar_army_sacking

	only_playable = yes
	#religion = catholic
	#min_age = 16
	
    trigger = {
		NOT = { has_global_flag = age_dark }
		year == 600
		independent = yes
		OR = {
			tier = DUKE
			tier = KING
			tier = EMPEROR
		}
		#controls_religion = yes # Let it happen for the Pope. He's always around.
    }

	immediate = {
		set_global_flag = age_dark
	}

	option = {
		name = "Civilisation is progressing..."
		if = {
			limit = {
				independent = yes
				OR = {
					tier = DUKE
					tier = KING
					tier = EMPEROR
				}
			}
			add_character_modifier = {
				name = age_dark_modifier
				years = 5
				#inherit = yes
				stacking = no
			}
		}
		custom_tooltip = {
			text = "The Prosperity of your capital decreases."
			hidden_tooltip = {
				capital_scope = { change_variable = { which = prosperity_value value = -50 } }
			}
		}
		prestige = -250
		piety = -250
	}
}
