    trigger = {
		has_character_flag = is_louis_pious
		NOT = { has_global_flag = hre_concluded }
		has_landed_title = e_hre
		start_date = 867.1.1
    }

	
	option = {
		name = EVTOPTHREAR.1
		ai_chance = { factor = 100 }
		unsafe_destroy_landed_title = e_hre
		set_global_flag = hre_concluded
		any_player = { 
			narrative_event = { id = HREAR.3 } 
		}
	}
	
	option = {
		name = EVTOPTHREAR.2
		ai_chance = { factor = 0 }
		set_global_flag = hre_concluded

			limit = {
				has_landed_title = e_hre
			}
			narrative_event = { id = HREAR.2 }
		}
		any_player = { 
			narrative_event = { id = HREAR.4 }
		}
	}
	
}

narrative_event = {
	id = HREAR.10
	title = HREARNAME10
	desc = HREDESC10
    is_triggered_only = yes

	picture = "GFX_evt_crusaders"
	border = GFX_event_narrative_frame_war
	
	option = {
		name = EVTOPTHREAR.4
		create_character = {
			name="Roderick"
			culture = norse
			religion = norse_pagan
			age = 32
			dynasty = random
			random_traits = yes
			attributes = {
				martial=9
				diplomacy=8
				intrigue=5
				stewardship=7
			}
		}
		
		new_character = {
			save_event_target_as = hre_invader_mission
			wealth = 1000
			prestige = 500
			piety = 500
			remove_trait = slow
			remove_trait = imbecile
			remove_trait = weak
			remove_trait = feeble
			character_event = { id = HREAR.11 }
		}
		
		259 = {
			kingdom = {
				event_target:hre_invader_mission = {
					unsafe_war = {
						casus_belli = scripted_invasion
						target = ROOT
						thirdparty_title = PREV
					}
				}
			}
		}
		any_playable_ruler = {
			limit = { ai = no }
			if = {
				limit = {
					OR = {
						has_power_omen_modifier_trigger = yes
						has_battle_omen_modifier_trigger = yes
					}
				}
				set_character_flag = negative_event_flag # For the scrying outcome calculation...
			}
		}
	}
}
