	trigger = {
		OR = {
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = random
				}
				total_years_played = 50
			}
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = delayed_random
				}
				year >= 1250
				total_years_played = 50
			}
			AND = {
				has_game_rule = {
					name = roman_invasion
					value = test
				}
				total_years_played <= 1
			}
		}
		NOT = { has_global_flag = roman_america_part_1_triggered }
		NOT = { has_global_flag = west_rome_formed }
		NOT = {
			has_game_rule = {
				name = roman_invasion
				value = off
			}
		}
		controls_religion = yes # Let it happen for the Pope. He's always around.
	}
	
	immediate = {
		any_independent_ruler = {
			limit = {
				OR = {
					culture = saxon
					culture = english
					culture = scottish
					culture = cumbric
					culture = anglonorse
				}
			}
			save_event_target_as = random_britanian_spotted_romans
		}
		any_player = { narrative_event = { id = ROMA.97 } }
		any_player = { set_character_flag = test_roman_americans_part_1 }
		set_global_flag = roman_america_part_1_triggered
		ROOT = { character_event = { id = ROMA.98 days = 1 } }
	}
