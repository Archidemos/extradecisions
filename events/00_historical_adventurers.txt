character_event = { #check start date
	id = HA.1
	hide_window = yes
	
	only_playable = yes
	
	trigger = {
		NOT = { start_date >= 936.1.1 }
		NOT = { has_global_flag = normans_spawned }
		trigger_if = {
			limit = { start_date == 867.1.1 }
			character = c_242
			c_242 = {
				is_alive = yes
				is_landed = no
				age >= 18
				is_incapable = no
			}
		}
		trigger_if = {
			limit = { start_date == 769.1.1 }
			year = 810
			culture = norse
			religion = norse_pagan
		}
	}
	
	mean_time_to_happen = {
		years = 10
	}
	
	immediate = {
		set_global_flag = normans_spawned
		if = {
			limit = { start_date == 867.1.1 }
			wealth = 200
			dynasty = 752
			any_child_even_if_dead = {
				dynasty = 752
			}
			add_trait = viking

				holder_scope = {
					narrative_event = { id = HA.2 }

				limit = {
					has_education_martial_trigger = no
				}

					5 = { add_trait = brilliant_strategist }
					8 = { add_trait = skilled_tactician }
					4 = { add_trait = tough_soldier }
					2 = { add_trait = misguided_warrior }
				}
			}
			add_trait = lunatic
			remove_trait = dull
			remove_trait = feeble
			remove_disease_trait_effect = yes
			save_event_target_as = rollo
		}
		else = {
			create_character = {
				random_traits = yes
				name = "Hrolfr"
				dynasty = 752

					martial = 8
					diplomacy = 8

				}
			}

				wealth = 200
				dynasty = 752
				any_child_even_if_dead = {
					dynasty = 752
				}
				add_trait = viking
				if = {
					limit = {
						has_education_martial_trigger = no
					}

						5 = { add_trait = brilliant_strategist }
						8 = { add_trait = skilled_tactician }

						2 = { add_trait = misguided_warrior }

        		}

			factor = 80
			modifier = {
				factor = 0

						has_landed_title = c_arques

		name = B_HA2
